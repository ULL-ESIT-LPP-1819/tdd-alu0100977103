    # File lib/gema/valoracion.rb, line 42
def valor_energe_kcal
    @kcal_hidrat=@hidratos_carb*4
    @kcal_prot=@prot*4
    @kcal_grasas=@cnt_grasas*9
    @kcal_sat=@cnt_grasas_sat*9
    @kcal_energia=0.0
    @kcal_energia=@kcal_energia+@kcal_grasas+@kcal_hidrat+@kcal_prot
end
