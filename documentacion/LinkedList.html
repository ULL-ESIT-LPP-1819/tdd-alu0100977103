def extract_by_begin()
    if @head == @tail
                    @head, @tail = nil
    else
            @head = @head[:next]
                    @tail = @tail[:prev]

    end

def insert_by_begin(value)
            if @head.nil?
                    @head = @Node.new(value, nil, nil)
                    @tail = @head
            else
                    @head[:prev] = @Node.new(value, @head, nil)
                    @head = @head[:prev]
            end
end

def insert_by_end(value)
            if @head.nil?
                    @head = @Node.new(value, nil, nil)
                    @tail = @head
            else
                    @tail[:next] = @Node.new(value, nil, @tail)
                    @tail = @tail[:next]
            end
end
