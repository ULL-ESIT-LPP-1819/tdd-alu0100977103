def circunferencia(value)
    if value==0
        vector=@cintura
    else
        vector=@cadera
    end
    
    @suma=0
    i=0
    while i !=vector.length
        @suma=@suma+vector[i]
        i=i+1
    end
    @suma=@suma/2.0
end

def plieges_cutaneos(value)
    if value==0
        vector=@tri
    elsif value==1
        vector=@bi
    elsif value==2
        vector=@su
    else
        vector=@sup
    end
    
    @suma=0
    i=0
    while i !=vector.length
        @suma=@suma+vector[i]
        i=i+1
    end
    @suma=@suma/3.0
end
